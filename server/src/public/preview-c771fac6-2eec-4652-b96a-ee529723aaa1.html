Here's an example of a beautiful and attractive button using React Native:

```jsx
// Button.js
import React from 'react';
import { View, Text, TouchableOpacity } from 'react-native';
import { Colors } from '../styles/Colors';
import { Fonts } from '../styles/Fonts';

const Button = ({ title, onPress, style, titleStyle }) => {
  return (
    <TouchableOpacity
      style={[
        {
          backgroundColor: Colors.primary,
          borderRadius: 10,
          paddingVertical: 15,
          paddingHorizontal: 20,
          elevation: 3,
          flexDirection: 'row',
          justifyContent: 'center',
          alignItems: 'center',
        },
        style,
      ]}
      onPress={onPress}
    >
      <Text style={[Fonts.h5, { color: Colors.white, ...titleStyle }]}>{title}</Text>
    </TouchableOpacity>
  );
};

export default Button;
```

And here's an example of how you can use the `Button` component in your app:

```jsx
// HomeScreen.js
import React from 'react';
import { View, Text } from 'react-native';
import Button from './Button';

const HomeScreen = () => {
  const handlePress = () => {
    console.log('Button pressed!');
  };

  return (
    <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
      <Button title="Press me!" onPress={handlePress} />
    </View>
  );
};

export default HomeScreen;
```

You can customize the styles and colors to fit your app's design.
